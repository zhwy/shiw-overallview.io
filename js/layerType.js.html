/**
 * 批量图层默认参数

 * @property {Function} batchModel 获取批量模型参数

const BatchLayer = {
  /**
   * 获取批量矢量图层参数
   * @param {Object} param
   * @param {String} param.name 图层名称
   * @param {String} param.code 项目编号
   * @param {String} param.parentId 父节点id
   * @param {Number} param.sort 排序
   * @returns {Object} layerBorder
   */
  batchVector: ({ name, code, parentId, sort }) => {
    let layerBorder = Default.vectorParams();
    layerBorder = {
      ...layerBorder,
      ...{
        name: name,
        layerId: '00000000-0000-0000-0000-000000000000',
        borderSize: 3,
        borderColor: Tool.randomColor(),
        borderType: 1,
        isGround: 2,
        remark: name
      }
    };
    return {
      id: '00000000-0000-0000-0000-000000000000',
      name: name,
      type: LayerType.VectorCollection,
      check: 1,
